    BegSr ReadFile 
        DclFld Counter Type(*Integer) Len(4)

        Counter = 0

        listboxStates.Items.Clear()

        Read Customer
        DoWhile (NOT Customer.IsEof)
            Counter = Counter + 1
            listboxStates.Items.Add(Customer_CMCity.Trim() + ", " + Customer_CMState)
            Read Customer
        EndDo 

        labelRecordsRead.Text = 'Records read: ' + Counter.ToString("#")
    EndSr
